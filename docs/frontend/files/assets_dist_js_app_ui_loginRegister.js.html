(function() {
  define([], function() {

    /**
     * Manage form tabs
     *
     * @class LoginRegister
     * @constructor
     */
    var LoginRegister;
    return LoginRegister = (function() {
      function LoginRegister(login, register, btn) {
        this.login = login;
        this.register = register;
        this.btn = btn;
        $(this.login.form).hide();
        $(this.register.form).hide();
        $(this.register.btn).addClass("btn " + this.btn.passive);
        $(this.login.btn).addClass("btn " + this.btn.passive);
        this;
      }


      /**
       * Show/hide forms depending on user actions
       *
       * @method toggleBtn
       */

      LoginRegister.prototype.toggleBtn = function() {
        $(this.register.form).hide();
        $(this.currentActive.btn).removeClass(this.btn.passive);
        $(this.currentActive.btn).addClass(this.btn.active);
        $(this.currentPassive.btn).removeClass(this.btn.active);
        $(this.currentPassive.btn).addClass(this.btn.passive);
        $(this.currentPassive.form).hide();
        if (!this.isActive()) {
          return $(this.currentActive.form).show();
        } else {
          $(this.currentActive.form).hide();
          $(this.currentActive.btn).removeClass(this.btn.active);
          $(this.currentActive.btn).addClass(this.btn.passive);
          return this.currentActive = {};
        }
      };


      /**
       * Compare the previous and current status
       * of the form in order to hide the active form
       * if it was previously actie
       *
       * @method isActive
       * @return {Boolean} isActive
       */

      LoginRegister.prototype.isActive = function() {
        return this.previousActive.form === this.currentActive.form;
      };


      /**
       *  Set the status of the form
       *
       *  @method setActive
       *  @param {String} active
       *  @return {Array} forms
       */

      LoginRegister.prototype.setActive = function(active) {
        this.previousActive = this.currentActive || {};
        if (active === "login") {
          this.currentActive = this.login;
          this.currentPassive = this.register;
        } else {
          this.currentActive = this.register;
          this.currentPassive = this.login;
        }
        return [this.currentActive, this.currentPassive];
      };

      return LoginRegister;

    })();
  });

}).call(this);
